    <script>
	
		
		// ************************************** //
		// ************************************** //
		// ************************************** //
		
		// Default setting initialization
		var full_extent = [0,0,10000,6923];
		var yuan_extent = [1695,2891,4217,5480];
		
		
		// Initializing the project
		var projection = new ol.proj.Projection({
		  code: 'xkcd-image',
		  units: 'pixels',
		  extent: full_extent
		});

		
		// ************************************** //
		// ************************************** //
		// ************************************** //
		
		// Style Declaration
		
		var style_city_capital = new ol.style.Style({
			image: new ol.style.Icon(({
				src: 'MapFiles/Icons/ico_city_1'
			}))
		});


		// ************************************** //
		// ************************************** //
		// ************************************** //
		
		// Icon Features //
		
		var iconFeature = new ol.Feature({
			geometry: new ol.geom.Point([3889, 5145]),
			name: 'Xiayang'
		});

		var vectorSource = new ol.source.Vector({
			features: [iconFeature]
			// ,projection: projection
		});

		var yuan_capital = new ol.layer.Vector({
			source: vectorSource
		});
		
		yuan_capital.setStyle(style_city_capital);

		  
		// ************************************** //
		// ************************************** //
		// ************************************** //
		
		// Defining the terrain 
		var map_yuan = new ol.layer.Image({
			  source: new ol.source.ImageStatic({
				url: 'MapFiles/Map_Yuan.png',
				projection: projection,
				imageExtent: yuan_extent
			  })
		})

		// ************************************** //
		// ************************************** //
		// ************************************** //
		
		// Final set of map initialization
		
		var map = new ol.Map({
		  layers: [map_yuan,yuan_capital],
		  target: 'map',
		  view: new ol.View({
			projection: projection,
			center: ol.extent.getCenter(full_extent),
			zoom: 3,
			maxZoom: 9,
			minZoom: 1
		  })
		});
		
    </script>
